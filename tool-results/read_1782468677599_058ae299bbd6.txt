  1588→  async function fetchTMDB(q) {
  1589→    const [movieRes, tvRes] = await Promise.all([
  1590→      fetch(`${CF_WORKER_URL}/3/search/movie?query=${encodeURIComponent(q)}&language=en-US&page=1`),
  1591→      fetch(`${CF_WORKER_URL}/3/search/tv?query=${encodeURIComponent(q)}&language=en-US&page=1`)
  1592→    ]);
  1593→
  1594→    const movieData = await movieRes.json();
  1595→    const tvData    = await tvRes.json();
  1596→
  1597→    const movieResults = (movieData.results || []).map(r => ({ ...r, media_type: 'movie' }));
  1598→    const tvResults    = (tvData.results    || []).map(r => ({ ...r, media_type: 'tv' }));
  1599→
  1600→    const combined = [...movieResults, ...tvResults]
  1601→      .sort((a, b) => (b.popularity || 0) - (a.popularity || 0))
  1602→      .slice(0, 6);
  1603→
  1604→    return combined.map(item => {
  1605→      const date      = item.release_date || item.first_air_date || '';
  1606→      const monthYear = date ? new Date(date).toLocaleDateString('en-US', { month: 'long', year: 'numeric' }) : '—';
  1607→      const typeLabel = item.media_type === 'movie' ? 'Movie' : 'TV';
  1608→      const poster    = item.poster_path ? `https://image.tmdb.org/t/p/w92${item.poster_path}` : '';
  1609→      const score     = item.vote_average ? `★ TMDB ${Number(item.vote_average).toFixed(1)}` : null;
  1610→      return {
  1611→        poster,
  1612→        title:    item.title || item.name || '—',
  1613→        original: item.original_title || item.original_name || '',
  1614→        meta:     `${monthYear} · ${typeLabel}`,
  1615→        score,
  1616→        id:       item.id,
  1617→        source:   'tmdb',
  1618→        mediaType: item.media_type
  1619→      };
  1620→    });
  1621→  }

  1623→  function renderSuggestions(results, q, container) {
  1624→    if (!results.length) {
  1625→      container.innerHTML = `<div style="padding:14px 12px;font-size:0.76rem;color:var(--text-muted,#888);">No results for "${esc(q)}"</div>`;
  1626→      return;
  1627→    }
  1628→
  1629→    const slugify = (title) => {
  1630→      if (!title) return '';
  1631→      return title
  1632→        .toLowerCase()
  1633→        .replace(/[^a-z0-9]+/g, '-')
  1634→        .replace(/^-+|-+$/g, '');
  1635→    };
  1636→
  1637→    const html = results.map(r => {
  1638→      let detailsUrl = '#';
  1639→      const slug = slugify(r.title);
  1640→
  1641→      if (r.source === 'jikan') {
  1642→        detailsUrl = `/info/anime/jikan-${r.id}`;
  1643→        if (slug) detailsUrl += `/${slug}`;
  1644→      } else if (r.source === 'tmdb') {
  1645→        const type = r.mediaType === 'movie' ? 'movie' : 'tv';
  1646→        const prefix = r.mediaType === 'movie' ? `tmdb-movie-${r.id}` : `tmdb-tv-${r.id}`;
  1647→        detailsUrl = `/info/${type}/${prefix}`;
  1648→        if (slug) detailsUrl += `/${slug}`;
  1649→      }
  1650→
  1651→      const img = r.poster
  1652→        ? `<img class="suggestion-poster" src="${esc(r.poster)}" alt="" loading="lazy" onerror="this.style.background='#1e2633';">`
  1653→        : `<div class="suggestion-poster"></div>`;
  1654→      const orig = r.original && r.original !== r.title
  1655→        ? `<div class="sug-orig">${esc(r.original)}</div>` : '';
  1656→      const score = r.score
  1657→        ? `<span class="sug-score">${esc(r.score)}</span>` : '';
  1658→      const meta = [r.meta, score].filter(Boolean).join(' · ');
  1659→
  1660→      return `<a href="${detailsUrl}" class="suggestion-item">
  1661→        ${img}
  1662→        <div class="suggestion-info">
  1663→          <div class="sug-title">${esc(r.title)}</div>
  1664→          ${orig}
  1665→          <div class="sug-meta">${meta}</div>
  1666→        </div>
  1667→      </a>`;
  1668→    }).join('');
  1669→
  1670→    container.innerHTML = html
  1671→      + `<button class="view-all-btn" onclick="location.href='/search?q=${encodeURIComponent(q)}&type=${currentSearchMode}'">${SVG.arrow} View all results</button>`;
  1672→    container.style.display = 'block';
  1673→  }

  1973→    return p.length >= 8 && /[A-Z]/.test(p) && /[a-z]/.test(p) && /[0-9]/.test(p) && /[^A-Za-z0-9]/.test(p);

  1976→  function updatePwdStrength(p) {
  1977→    let score = 0;
  1978→    if (p.length >= 8) score++;
  1979→    if (/[A-Z]/.test(p)) score++;
  1980→    if (/[a-z]/.test(p)) score++;
  1981→    if (/[0-9]/.test(p)) score++;
  1982→    if (/[^A-Za-z0-9]/.test(p)) score++;
